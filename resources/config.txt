# MyPrtSc的配置文件
# 修改配置后请重启程序。 如遇程序启动失败，请删除此文件，软件将自动恢复默认值

# 截图保存的根目录
BaseDir = D:\MyPrtSc_screenshot

# 是否转换窗口名称中的日文乱码
IfAutoConvert = true

# 若游戏为窗口运行，是否仅截取该窗口
IfWindowShot = true

# 图片存储配置，有如下几种：
# PNG	-- 默认值。Windows API原生截图，巨大。
# PNG+	-- 使用optipng进行png优化，较大但无损。更高的CPU占用，更慢见到结果。
# JPG80	-- 保存为jpg，较小但有损。数字"80"表示图片质量，可取0-100。
# AVIF80	-- 保存为avif，微小但有损。数字"80"表示图片质量，可取0-100。
Format = JPG80

# 截图热键设置，支持键盘上的所有单个按键
# 普通字符, SPACE, UP, DOWN, LEFT, RIGHT, INSERT, DELETE, HOME, END, PAGEUP, PAGEDOWN, SHIFT, ALT, CTRL, CAPSLOCK, ESC, TAB, ENTER, BACKSPACE, WIN
HOTKEY_A = PRTSC
